let peer; let streamDeVideo; function mostrarTela(id) { document.querySelectorAll('.tela').forEach(t => t.classList.remove('ativa')); document.getElementById(id).classList.add('ativa'); } // Cria um "Código Único" juntando o nome e a senha function gerarIdDaSala(nome, senha) { return "cine-" + nome.toLowerCase().trim() + "-" + senha.trim(); } // ---------------------------------------------------- // LÓGICA DO HOST (QUEM CRIA A SALA E COMPARTILHA A TELA) // ---------------------------------------------------- async function criarSala() { const nome = document.getElementById('criar-nome').value; const senha = document.getElementById('criar-senha').value; const status = document.getElementById('status-host'); if(!nome || !senha) return alert("Preencha o nome e a senha!"); const salaId = gerarIdDaSala(nome, senha); status.innerText = "Pedindo permissão de tela..."; try { // Pede a tela para o usuário (funciona no PC) streamDeVideo = await navigator.mediaDevices.getDisplayMedia({ video: true, audio: true // IMPORTANTE: ao escolher a tela, o usuário deve marcar "Compartilhar áudio do sistema" }); // Cria o peer na nuvem com o ID exato da sala e senha peer = new Peer(salaId); peer.on('open', (id) => { mostrarTela('cinema'); document.getElementById('titulo-cinema').innerText = "Transmissão da Sala: " + nome; const videoElement = document.getElementById('videoPlayer'); videoElement.srcObject = streamDeVideo; videoElement.muted = true; // O Host não precisa ouvir a si mesmo }); // Quando alguém (Mobile) entrar, o Host atende a chamada e envia a tela peer.on('call', (chamada) => { console.log("Novo espectador conectado!"); chamada.answer(streamDeVideo); // Envia o filme! }); // Se a transmissão parar streamDeVideo.getVideoTracks()[0].onended = () => { alert("Transmissão encerrada."); location.reload(); }; } catch (err) { status.innerText = "Erro: " + err.message; console.error(err); } } // ---------------------------------------------------- // LÓGICA DO ESPECTADOR (QUEM VAI ASSISTIR NO MOBILE/PC) // ---------------------------------------------------- function entrarSala() { const nome = document.getElementById('entrar-nome').value; const senha = document.getElementById('entrar-senha').value; const status = document.getElementById('status-viewer'); if(!nome || !senha) return alert("Preencha o nome e a senha!"); const salaId = gerarIdDaSala(nome, senha); status.innerText = "Conectando..."; // Cria um peer sem ID (ID aleatório para quem assiste) peer = new Peer(); peer.on('open', (id) => { // Liga para o Host usando o ID da sala const chamada = peer.call(salaId); chamada.on('stream', (streamRemoto) => { // Recebeu o vídeo! mostrarTela('cinema'); document.getElementById('titulo-cinema').innerText = "Assistindo Sala: " + nome; const videoElement = document.getElementById('videoPlayer'); videoElement.srcObject = streamRemoto; // Tenta dar play automaticamente videoElement.play().catch(e => { console.log("Play automático bloqueado. O usuário precisa clicar no player."); }); }); chamada.on('close', () => { alert("A transmissão acabou."); location.reload(); }); }); peer.on('error', (err) => { if(err.type === 'peer-unavailable') { status.innerText = "Sala não encontrada ou senha incorreta!"; } else { status.innerText = "Erro ao conectar."; } }); }
